layui.use(['notice'], function () {

    // 全部参数都是可选参数
    var notice = {
        type: "",
        calssName: "",
        title: "创建普通notice通知",
        icon: "",
        align: "",
        autoClose: true,
        time: 3000,
        click: true,  
        desktop: true, // 开启桌面提醒    
        end: function (){ 

        }
    };
    notice.init(notice);
});


// 关闭全部
notice.closeAll();
